# 프로젝트 생성 및 기본 구성

$ laravel new laravel-movie-app

$ cd laravel-movie-app


# tailwindcss 설정 (참고 https://tailwindcss.com/docs/guides/laravel)

$ npm install -D tailwindcss@latest postcss@latest autoprefixer@latest

$ npx tailwindcss init

$ npm run dev


$ php artisan serve
